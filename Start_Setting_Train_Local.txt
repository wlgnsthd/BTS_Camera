sudo apt-get update
sudo apt-get install git cmake libpython3-dev python3-numpy python3-pip
git clone --recursive https://github.com/dusty-nv/jetson-inference
cd jetson-inference

cd build
cmake ../
#check Pytorch *

cd jetson-inference/build
./install-pytorch.sh
